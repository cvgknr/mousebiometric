DROP TABLE MS_FEATURE;
DROP TABLE MS_USER;
DROP TABLE MS_FILE;
CREATE TABLE MS_USER(ID INT PRIMARY KEY, first_name VARCHAR(100),last_name VARCHAR(100));
CREATE TABLE MS_FEATURE(ID INT PRIMARY KEY, value VARCHAR(100),classname VARCHAR(100),unit VARCHAR(100),label VARCHAR(100),user_id INT);
ALTER TABLE MS_FEATURE  ADD FOREIGN KEY (USER_ID) REFERENCES MS_USER (id);
CREATE TABLE MS_FILE(ID INT PRIMARY KEY, checksum VARCHAR(100),user_id INT);
ALTER TABLE MS_FILE  ADD FOREIGN KEY (USER_ID) REFERENCES MS_USER (id);
